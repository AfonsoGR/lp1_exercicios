using System;

namespace repo_exercicios
{
    class Program
    {

        static void Main(string[] args)
        {
            // Objetos de classe Line
            Line linha1 = null;

            Line linha2 = null;

            //Variáveis do input do util
            //Que iram definir as linhas
            //através das coordenadas pedidas mais abaixo
            double xi, yi, xf, yf;

            //Loop que vai perguntando coordenadas ao utilizador até o
            //o programa ter suficientes para formar 2 linhas
            while (true)
            {
                Console.WriteLine("Please input the coordinates of one " +
                    "point belonging to the new line you want to make:");

                Console.WriteLine("Input X: ");
                xi = Double.Parse(Console.ReadLine());
                Console.WriteLine("Input Y: ");
                yi = Double.Parse(Console.ReadLine());

                Console.WriteLine("\nNow do the same for another point " +
                    "in that line");

                Console.WriteLine("Input X: ");
                xf = Double.Parse(Console.ReadLine());
                Console.WriteLine("Input Y: ");
                yf = Double.Parse(Console.ReadLine());

                //Verifica que objeto de Line é que deve incializar 
                //e inicialisa o correto
                if (linha1 == null)
                    linha1 = new Line(xi, yi, xf, yf);

                else if (linha1 != null && linha2 == null)
                {
                    linha2 = new Line(xi, yi, xf, yf);
                    break;
                }

            }

            //Verifica se as linhas se cruzam usando o método dado

            if (Cross(linha1))
                Console.WriteLine("The lines you defined intersect");
            else
                Console.WriteLine("The lines you defined do not intersect");

        }
    }
}
