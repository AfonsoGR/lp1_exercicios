```cs
namespace repo_exercicios

    /// <summary>
    /// Tipos de NPC, cada elemento da enumeração foi-lhe atribuido 
    /// (em ordem descendente) o valor binário de 001, 010 e 100

    enum NPCType
    {

        Minion = 1 << 0,
        Soldier = 1 << 1,
        Boss = 1 << 2,

    }

}
```

```cs
//NPC.cs

using System;

namespace repo_exercicios
{
    class NPC
    {
        private float _energy;
        private float _damage;
        private float _speed;

        private NPCType _npcType;

        /// <summary>
        /// Construtor da Class NPC
        /// </summary>
        /// <param name="energy">Energia/Vitalidade deste NPC</param>
        /// <param name="damage">Valor base para o dano que este NPC causa nos 
        /// seus ataques </param>
        /// <param name="speed">Velocidade do NPC</param>
        /// <param name="type">Tipo de NPC, influencia o dano que ele 
        /// causa</param>
        public NPC(float energy, float damage, float speed, NPCType type)
        {
            _energy = energy;
            _damage = damage;
            _speed = speed;
            _npcType = type;

        }

        //Põe a energia do NPC a metade do que era antes
        public float TakeHit() => _energy /= 2;

        //Põe energia e velocidade a 0, NPC morreu 
        public void Die()
        {

            _energy = 0;
            _speed = 0;

        }

        //Aumenta e diminui a velocidade do NPC por 10%, respetivamente.
        public float Faster() => _speed += _speed * 0.1f;

        public float Slower() => _speed -= _speed * 0.1f;


        /* 
         
        Retorna o valor de _damage multiplicada pêlo valor binário do
        elemento da enumeração do seu type:
        
        (int)_npcType retorna o valor numérico desse elemento da 
        enumeração
         
        Convert.ToString((int)_npcType, 2) retorna, como string, o valor 
        binário do inteiro retornado anteriormente

        Finalmente, utilizando o método Int32.Parse(), convertemos o 
        string num número inteiro para o podermos usar na multiplicação

        */
        public float Hit()
            => _damage *= Int32.Parse(Convert.ToString((int)_npcType, 2));

    }
}


```

```cs
//Program.cs

using System;

namespace repo_exercicios
{
    class Program
    {
        static void Main(string[] args)
        {

            NPC William = new NPC(12, 50, 100, NPCType.Boss);
            Console.WriteLine($"Slow down: {William.Slower()}");
            Console.WriteLine($"Speed Up: {William.Faster()}");
            Console.WriteLine($"Hit me for {William.Hit()} damage");
            Console.WriteLine($"I hit! Now you have: {William.TakeHit()} hp");
            William.Die();
            Console.WriteLine($"RIP");

            NPC Jimmy = new NPC(25, 5, 100, NPCType.Minion);
            Console.WriteLine($"Speed Up: {Jimmy.Faster()}");
            Console.WriteLine($"Slow up MORE: {Jimmy.Faster()}");
            Console.WriteLine($"Hit me for {Jimmy.Hit()} damage");
            Console.WriteLine($"I hit! Now you have: {Jimmy.TakeHit()} hp");
            Jimmy.Die();
            Console.WriteLine($"RIP");

        }
    }
}
```
